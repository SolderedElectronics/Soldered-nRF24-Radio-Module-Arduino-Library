examples/AcknowledgementPayloads/AcknowledgementPayloads.ino

Written by 2bndy5 in 2020

A simple example of sending data from 1 nRF24L01 transceiver to another with Acknowledgement (ACK) payloads attached to ACK packets.

This example was written to be used on 2 devices acting as "nodes". Use the Serial Monitor to change each node's behavior.

21#include "nRF24-SOLDERED.h"
22
23// instantiate an object for the nRF24L01 transceiver
24nRF24 radio(7, 8); // using pin 7 for the CE pin, and pin 8 for the CSN pin
25
26// an identifying device destination
27// Let these addresses be used for the pair
28uint8_t address[][6] = {"1Node", "2Node"};
29// It is very helpful to think of an address as a path instead of as
30// an identifying device destination
31// to use different addresses on a pair of radios, we need a variable to
32
33// uniquely identify which address this radio will use to transmit
34bool radioNumber = 1; // 0 uses address[0] to transmit, 1 uses address[1] to transmit
35
36// Used to control whether this node is sending or receiving
37bool role = false; // true = TX role, false = RX role
38
39// For this example, we'll be using a payload containing
40// a string & an integer number that will be incremented
41// on every successful transmission.
42// Make a data structure to store the entire payload of different datatypes
43struct PayloadStruct
44{
45 char message[7]; // only using 6 characters for TX & ACK payloads
46 uint8_t counter;
47};
48PayloadStruct payload;
49
50void setup()
51{
52
53 Serial.begin(115200);
54 while (!Serial)
55 {
56 // some boards need to wait to ensure access to serial over USB
57 }
58
59 // initialize the transceiver on the SPI bus
60 if (!radio.begin())
61 {
62 Serial.println(F("radio hardware is not responding!!"));
63 while (1)
64 {
65 } // hold in infinite loop
66 }
67
68 // print example's introductory prompt
69 Serial.println(F("RF24/examples/AcknowledgementPayloads"));
70
71 // To set the radioNumber via the Serial monitor on startup
72 Serial.println(F("Which radio is this? Enter '0' or '1'. Defaults to '0'"));
73 while (!Serial.available())
74 {
75 // wait for user input
76 }
77 char input = Serial.parseInt();
78 radioNumber = input == 1;
79 Serial.print(F("radioNumber = "));
80 Serial.println((int)radioNumber);
81
82 // role variable is hardcoded to RX behavior, inform the user of this
83 Serial.println(F("*** PRESS 'T' to begin transmitting to the other node"));
84
85 // Set the PA Level low to try preventing power supply related problems
86 // because these examples are likely run with nodes in close proximity to
87 // each other.
88 radio.setPALevel(RF24_PA_LOW); // RF24_PA_MAX is default.
89
90 // to use ACK payloads, we need to enable dynamic payload lengths (for all nodes)
91 radio.enableDynamicPayloads(); // ACK payloads are dynamically sized
92
93 // Acknowledgement packets have no payloads by default. We need to enable
94 // this feature for all nodes (TX & RX) to use ACK payloads.
95 radio.enableAckPayload();
96
97 // set the TX address of the RX node into the TX pipe
98 radio.openWritingPipe(address[radioNumber]); // always uses pipe 0
99
100 // set the RX address of the TX node into a RX pipe
101 radio.openReadingPipe(1, address[!radioNumber]); // using pipe 1
102
103 // additional setup specific to the node's role
104 if (role)
105 {
106 // setup the TX payload
107
108 memcpy(payload.message, "Hello ", 6); // set the payload message
109 radio.stopListening(); // put radio in TX mode
110 }
111 else
112 {
113 // setup the ACK payload & load the first response into the FIFO
114
115 memcpy(payload.message, "World ", 6); // set the payload message
116 // load the payload for the first received transmission on pipe 0
117 radio.writeAckPayload(1, &payload, sizeof(payload));
118
119 radio.startListening(); // put radio in RX mode
120 }
121
122 // For debugging info
123 // printf_begin(); // needed only once for printing details
124 // radio.printDetails(); // (smaller) function that prints raw register values
125 // radio.printPrettyDetails(); // (larger) function that prints human readable data
126}

128void loop()
129{
130
131 if (role)
132 {
133 // This device is a TX node
134
135 unsigned long start_timer = micros(); // start the timer
136 bool report = radio.write(&payload, sizeof(payload)); // transmit & save the report
137 unsigned long end_timer = micros(); // end the timer
138
139 if (report)
140 {
141 Serial.print(F("Transmission successful! ")); // payload was delivered
142 Serial.print(F("Time to transmit = "));
143 Serial.print(end_timer - start_timer); // print the timer result
144 Serial.print(F(" us. Sent: "));
145 Serial.print(payload.message); // print the outgoing message
146 Serial.print(payload.counter); // print the outgoing counter
147 uint8_t pipe;
148 if (radio.available(&pipe))
149 { // is there an ACK payload? grab the pipe number that received it
150 PayloadStruct received;
151 radio.read(&received, sizeof(received)); // get incoming ACK payload
152 Serial.print(F(" Recieved "));
153 Serial.print(radio.getDynamicPayloadSize()); // print incoming payload size
154 Serial.print(F(" bytes on pipe "));
155 Serial.print(pipe); // print pipe number that received the ACK
156 Serial.print(F(": "));
157 Serial.print(received.message); // print incoming message
158 Serial.println(received.counter); // print incoming counter
159
160 // save incoming counter & increment for next outgoing
161 payload.counter = received.counter + 1;
162 }
163 else
164 {
165 Serial.println(F(" Recieved: an empty ACK packet")); // empty ACK packet received
166 }
167 }
168 else
169 {
170 Serial.println(F("Transmission failed or timed out")); // payload was not delivered
171 }
172
173 // to make this example readable in the serial monitor
174 delay(1000); // slow transmissions down by 1 second
175 }
176 else
177 {
178 // This device is a RX node
179
180 uint8_t pipe;
181 if (radio.available(&pipe))
182 { // is there a payload? get the pipe number that recieved it
183 uint8_t bytes = radio.getDynamicPayloadSize(); // get the size of the payload
184 PayloadStruct received;
185 radio.read(&received, sizeof(received)); // get incoming payload
186 Serial.print(F("Received "));
187 Serial.print(bytes); // print the size of the payload
188 Serial.print(F(" bytes on pipe "));
189 Serial.print(pipe); // print the pipe number
190 Serial.print(F(": "));
191 Serial.print(received.message); // print incoming message
192 Serial.print(received.counter); // print incoming counter
193 Serial.print(F(" Sent: "));
194 Serial.print(payload.message); // print outgoing message
195 Serial.println(payload.counter); // print outgoing counter
196
197 // save incoming counter & increment for next outgoing
198 payload.counter = received.counter + 1;
199 // load the payload for the first received transmission on pipe 0
200 radio.writeAckPayload(1, &payload, sizeof(payload));
201 }
202 } // role
203
204 if (Serial.available())
205 {
206 // change the role via the serial monitor
207
208 char c = toupper(Serial.read());
209 if (c == 'T' && !role)
210 {
211 // Become the TX node
212
213 role = true;
214 Serial.println("* CHANGE TO TRANSMIT ROLE -- PRESS 'R' TO SWITCH BACK");
215
216 memcpy(payload.message, "Hello ", 6); // change payload message
217 radio.stopListening(); // this also discards any unused ACK payloads
218 }
219 else if (c == 'R' && role)
220 {
221 // Become the RX node
222
223 role = false;
224 Serial.println(F("* CHANGE TO RECEIVE ROLE -- PRESS 'T' TO SWITCH BACK"));
225 memcpy(payload.message, "World ", 6); // change payload message
226
227 // load the payload for the first received transmission on pipe 0
228 radio.writeAckPayload(1, &payload, sizeof(payload));
229 radio.startListening();
230 }
231 }
232} // loop
