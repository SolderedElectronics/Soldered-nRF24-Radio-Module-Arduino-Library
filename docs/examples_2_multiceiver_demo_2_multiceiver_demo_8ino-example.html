examples/MulticeiverDemo/MulticeiverDemo.ino

Written by 2bndy5 in 2020

A simple example of sending data from as many as 6 nRF24L01 transceivers to 1 receiving transceiver. This technique is trademarked by Nordic Semiconductors as "MultiCeiver".

This example was written to be used on up to 6 devices acting as TX nodes & only 1 device acting as the RX node (that's a maximum of 7 devices). Use the Serial Monitor to change each node's behavior.

23#include "nRF24-SOLDERED.h"
24
25// instantiate an object for the nRF24L01 transceiver
26nRF24 radio(7, 8); // using pin 7 for the CE pin, and pin 8 for the CSN pin
27
28// For this example, we'll be using 6 addresses; 1 for each TX node
29// It is very helpful to think of an address as a path instead of as
30// an identifying device destination
31// Notice that the last byte is the only byte that changes in the last 5
32// addresses. This is a limitation of the nRF24L01 transceiver for pipes 2-5
33// because they use the same first 4 bytes from pipe 1.
34uint64_t address[6] = {0x7878787878LL, 0xB3B4B5B6F1LL, 0xB3B4B5B6CDLL, 0xB3B4B5B6A3LL, 0xB3B4B5B60FLL, 0xB3B4B5B605LL};
35
36// Because this example allow up to 6 nodes (specified by numbers 0-5) to
37// transmit and only 1 node to receive, we will use a negative value in our
38// role variable to signify this node is a receiver.
39// role variable is used to control whether this node is sending or receiving
40char role = 'R'; // integers 0-5 = TX node; character 'R' or integer 82 = RX node
41
42// For this example, we'll be using a payload containing
43// a node ID number and a single integer number that will be incremented
44// on every successful transmission.
45// Make a data structure to use as a payload.
46struct PayloadStruct
47{
48 unsigned long nodeID;
49 unsigned long payloadID;
50};
51PayloadStruct payload;
52
53// This example uses all 6 pipes to receive while TX nodes only use 2 pipes
54// To make this easier we'll use a function to manage the addresses, and the
55// payload's nodeID
56void setRole(); // declare a prototype; definition is found after the loop()
57

59{
60
61 Serial.begin(115200);
62 while (!Serial)
63 {
64 // some boards need to wait to ensure access to serial over USB
65 }
66
67 // initialize the transceiver on the SPI bus
68 if (!radio.begin())
69 {
70 Serial.println(F("radio hardware is not responding!!"));
71 while (1)
72 {
73 } // hold in infinite loop
74 }
75
76 // print example's introductory prompt
77 Serial.println(F("RF24/examples/MulticeiverDemo"));
78 Serial.println(F("*** Enter a number between 0 and 5 (inclusive) to change"));
79 Serial.println(F(" the identifying node number that transmits."));
80
81 // Set the PA Level low to try preventing power supply related problems
82 // because these examples are likely run with nodes in close proximity of
83 // each other.
84 radio.setPALevel(RF24_PA_LOW); // RF24_PA_MAX is default.
85
86 // save on transmission time by setting the radio to only transmit the
87 // number of bytes we need to transmit a float
88 radio.setPayloadSize(sizeof(payload)); // 2x int datatype occupy 8 bytes
89
90 // Set the pipe addresses accordingly. This function additionally also
91 // calls startListening() or stopListening() and sets the payload's nodeID
92 setRole();
93
94 // For debugging info
95 // printf_begin(); // needed only once for printing details
96 // radio.printDetails(); // (smaller) function that prints raw register values
97 // radio.printPrettyDetails(); // (larger) function that prints human readable data
98
99} // setup()

101void loop()
102{
103
104 if (role <= 53)
105 {
106 // This device is a TX node
107
108 unsigned long start_timer = micros(); // start the timer
109 bool report = radio.write(&payload, sizeof(payload)); // transmit & save the report
110 unsigned long end_timer = micros(); // end the timer
111
112 if (report)
113 {
114 // payload was delivered
115
116 Serial.print(F("Transmission of payloadID "));
117 Serial.print(payload.payloadID); // print payloadID
118 Serial.print(F(" as node "));
119 Serial.print(payload.nodeID); // print nodeID
120 Serial.print(F(" successful!"));
121 Serial.print(F(" Time to transmit: "));
122 Serial.print(end_timer - start_timer); // print the timer result
123 Serial.println(F(" us"));
124 }
125 else
126 {
127 Serial.println(F("Transmission failed or timed out")); // payload was not delivered
128 }
129 payload.payloadID++; // increment payload number
130
131 // to make this example readable in the serial monitor
132 delay(1000); // slow transmissions down by 1 second
133 }
134 else if (role == 'R')
135 {
136 // This device is the RX node
137
138 uint8_t pipe;
139 if (radio.available(&pipe))
140 { // is there a payload? get the pipe number that recieved it
141 uint8_t bytes = radio.getPayloadSize(); // get the size of the payload
142 radio.read(&payload, bytes); // fetch payload from FIFO
143 Serial.print(F("Received "));
144 Serial.print(bytes); // print the size of the payload
145 Serial.print(F(" bytes on pipe "));
146 Serial.print(pipe); // print the pipe number
147 Serial.print(F(" from node "));
148 Serial.print(payload.nodeID); // print the payload's origin
149 Serial.print(F(". PayloadID: "));
150 Serial.println(payload.payloadID); // print the payload's number
151 }
152 } // role
153
154 if (Serial.available())
155 {
156 // change the role via the serial monitor
157
158 char c = Serial.read();
159 if (toupper(c) == 'R' && role <= 53)
160 {
161 // Become the RX node
162
163 role = 'R';
164 Serial.println(F("*** CHANGING ROLE TO RECEIVER ***"));
165 Serial.println(F("--- Enter a number between 0 and 5 (inclusive) to act as"));
166 Serial.println(F(" a unique node number that transmits to the RX node."));
167 setRole(); // change address on all pipes to TX nodes
168 }
169 else if (c >= 48 && c <= 53 && c != role)
170 {
171 // Become a TX node with identifier 'c'
172
173 role = c - 48;
174 Serial.print(F("*** CHANGING ROLE TO NODE "));
175 Serial.print(c);
176 Serial.println(F(" ***"));
177 Serial.println(F("--- Enter a number between 0 and 5 (inclusive) to change"));
178 Serial.println(F(" the identifying node number that transmits."));
179 Serial.println(F("--- PRESS 'R' to act as the RX node."));
180 setRole(); // change address on pipe 0 to the RX node
181 }
182 }
183
184} // loop

186void setRole()
187{
188 if (role == 'R')
189 {
190 // For the RX node
191
192 // Set the addresses for all pipes to TX nodes
193 for (uint8_t i = 0; i < 6; ++i)
194 radio.openReadingPipe(i, address[i]);
195
196 radio.startListening(); // put radio in RX mode
197 }
198 else
199 {
200 // For the TX node
201
202 // set the payload's nodeID & reset the payload's identifying number
203 payload.nodeID = role;
204 payload.payloadID = 0;
205
206 // Set the address on pipe 0 to the RX node.
207 radio.stopListening(); // put radio in TX mode
208 radio.openWritingPipe(address[(int)role]);
209
210 // According to the datasheet, the auto-retry features's delay value should
211 // be "skewed" to allow the RX node to receive 1 transmission at a time.
212 // So, use varying delay between retry attempts and 15 (at most) retry attempts
213 radio.setRetries(((role * 3) % 12) + 3, 15); // maximum value is 15 for both args
214 }
215} // setRole
